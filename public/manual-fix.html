-- Replace 'your-email@example.com' with your actual email
UPDATE profiles 
SET 
    is_paid = true,
    is_active = true,
    subscription_plan = 'monthly',
    updated_at = NOW()
WHERE email = 'your-email@example.com';

-- If no rows were affected, create a new profile:
-- First, get your user ID from auth.users:
SELECT id, email FROM auth.users WHERE email = 'your-email@example.com';

-- Then insert a new profile (replace USER_ID_HERE with the ID from above):
INSERT INTO profiles (
    id, email, is_paid, is_active, subscription_plan, 
    created_at, updated_at
) VALUES (
    'USER_ID_HERE', 'your-email@example.com', true, true, 'monthly',
    NOW(), NOW()
) ON CONFLICT (id) DO UPDATE SET
    is_paid = true,
    is_active = true,
    updated_at = NOW();
